\section{User Manual}

This section gives an exhaustive presentation of the objects and functions provided by the $MultivariateRandomMixture$ module.

\subsection{PythonMultivariateRandomMixture}


\begin{description}
\item[Usage :]  \strut
  \begin{description}
  \item $PythonMultivariateRandomMixture(collection, matrix)$
  \item $PythonMultivariateRandomMixture(collection, matrix, vector)$
  \end{description}
\item[Arguments :] \strut
  \begin{description}
  \item $collection$: a DistributionCollection, a collection of OpenTURNS distributions.
  \item $matrix$: a Matrix, the matrix of the affine transformation.
  \item $vector$: a NumericalPoint, the constant term of the affine transformation.
  \end{description}
\item[Some methods :]  \strut

  \begin{description}

  \item $computeCharacteristicFunction$
    \begin{description}
    \item[Usage :]  $computeCharacteristicFunction(u)$
    \item[Arguments :]  \strut
    \begin{description}
      \item $u$ : a NumericalPoint of dimension $n \ge 1$
    \end{description}
    \item[Value :] a $NumericalComplex$
    \end{description}

  \item $computePDF$
    \begin{description}
    \item[Usage :]  $computePDF(u)$
    \item[Arguments :]  \strut
    \begin{description}
      \item $u$ : a NumericalPoint of dimension $n \ge 1$
    \end{description}
    \item[Value :] a $NumericalScalar$
    \end{description}

  \item $getConstant$
    \begin{description}
    \item[Usage :]  $getConstant()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $NumericalPoint$, the deterministic vector of the transformation
    \end{description}

  \item $getCovariance$
    \begin{description}
    \item[Usage :]  $getCovariance()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $CovarianceMatrix$, the covariance of the distribution
    \end{description}

  \item $getDistributionCollection$
    \begin{description}
    \item[Usage :]  $getDistributionCollection()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $DistributionCollection$, the collection of distributions used for the affine transformation
    \end{description}

  \item $getMatrix$
    \begin{description}
    \item[Usage :]  $getMatrix()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $Matrix$, the <<~weight~>> matrix used for the affine transformation
    \end{description}

  \item $getMean$
    \begin{description}
    \item[Usage :]  $getMean()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $NumericalPoint$, the mean of the distribution
    \end{description}

  \item $getRange$
    \begin{description}
    \item[Usage :]  $getRangee()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] an $Interval$ of dimension $d$, the range of the distribution
    \end{description}

  \item $getRealization$
    \begin{description}
    \item[Usage :]  $getRealization()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $NumericalPoint$, a realization the distribution
    \end{description}

  \item $getSample$
    \begin{description}
    \item[Usage :]  $getSample()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $n$ : a positive integer
    \end{description}
    \item[Value :] a $NumericalSample$, a sample of $n$ realizations of the distribution
    \end{description}

  \item $getStandardDeviation$
    \begin{description}
    \item[Usage :]  $getStandardDeviation()$
    \item[Arguments :]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value :] a $NumericalPoint$, the standard deviation of the distribution
    \end{description}

  \end{description}

\end{description}
