\section{User Manual}

This section gives an exhaustive presentation of the objects and functions provided by the $MultivariateRandomMixture$ module.

\subsection{PythonMultivariateRandomMixture}


\begin{description}
\item[Usage:]  \strut
  \begin{description}
  \item $PythonMultivariateRandomMixture(collection, matrix)$
  \item $PythonMultivariateRandomMixture(collection, matrix, vector)$
  \end{description}
\item[Arguments:] \strut
  \begin{description}
  \item $collection$: a DistributionCollection, a collection of OpenTURNS distributions.
  \item $matrix$: a Matrix, the matrix of the affine transformation.
  \item $vector$: a NumericalPoint, the constant term of the affine transformation.
  \end{description}
\item[Some methods:]  \strut

  \begin{description}

  \item $computeCharacteristicFunction$
    \begin{description}
    \item[Usage:]  $computeCharacteristicFunction(u)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $u$: a NumericalPoint of dimension $n \ge 1$
    \end{description}
    \item[Value:] a $NumericalComplex$, compute the characteristic function on $u$
    \end{description}

  \item $computeLogCharacteristicFunction$
    \begin{description}
    \item[Usage:]  $computeLogCharacteristicFunction(u)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $u$: a NumericalPoint of dimension $n \ge 1$
    \end{description}
    \item[Value:] a $NumericalComplex$, compute the log-characteristic function on $u$
    \end{description}

  \item $computePDF$
    \begin{description}
    \item[Usage:]  $computePDF(u)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $u$: a NumericalPoint of dimension $n \ge 1$
    \end{description}
    \item[Value:] a $NumericalScalar$, compute the probability density function on $u$
    \end{description}

  \item $getAlpha$
    \begin{description}
    \item[Usage:]  $getAlpha()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalScalar$, the $\alpha$ parameter used for the computation of the range (position indicator)
    \end{description}

  \item $getBeta$
    \begin{description}
    \item[Usage:]  $getBeta()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalScalar$, the $\beta$ parameter used for the computation of the range (dispersion indicator)
    \end{description}

  \item $getBlockMax$
    \begin{description}
    \item[Usage:]  $getBlockMax()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] an $UnsignedLong$, the maximal parameter size (i.e $2^{blocMax}$) used for the computation of the characteristic function differences
    \end{description}

  \item $getBlockMin$
    \begin{description}
    \item[Usage:]  $getBlockMin()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] an $UnsignedLong$, the minimal parameter size (i.e $2^{blocMin}$) used for the computation of the characteristic function differences
    \end{description}

  \item $getConstant$
    \begin{description}
    \item[Usage:]  $getConstant()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalPoint$, the deterministic vector of the transformation
    \end{description}

  \item $getCorrelation$
    \begin{description}
    \item[Usage:]  $getCorrelation()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $CorrelationMatrix$, the correlation of the distribution
    \end{description}

  \item $getCovariance$
    \begin{description}
    \item[Usage:]  $getCovariance()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $CovarianceMatrix$, the covariance of the distribution
    \end{description}

  \item $getDimension$
    \begin{description}
    \item[Usage:]  $getDimension()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] an $UnsignedLong$, the dimension of the matrix
    \end{description}

  \item $getDistributionCollection$
    \begin{description}
    \item[Usage:]  $getDistributionCollection()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $DistributionCollection$, the collection of distributions used for the affine transformation
    \end{description}

  \item $getLastPDFError$
    \begin{description}
    \item[Usage:]  $getLastPDFError()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalScalar$, the last error obtained when computing the probability density function
    \end{description}

  \item $getMatrix$
    \begin{description}
    \item[Usage:]  $getMatrix()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $Matrix$, the <<~weight~>> matrix used for the affine transformation
    \end{description}

  \item $getMaxSize$
    \begin{description}
    \item[Usage:]  $getMaxSize()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] an $UnsignedLong$, the maximum size of the cache for the CharacteristicFunction values
    \end{description}

  \item $getMean$
    \begin{description}
    \item[Usage:]  $getMean()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalPoint$, the mean of the distribution
    \end{description}

  \item $getPDFPrecision$
    \begin{description}
    \item[Usage:]  $getPDFPrecision()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalScalar$, the precision used for the computation of the probability density function
    \end{description}

  \item $getRange$
    \begin{description}
    \item[Usage:]  $getRange()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] an $Interval$ of dimension $d$, the range of the distribution
    \end{description}

  \item $getRealization$
    \begin{description}
    \item[Usage:]  $getRealization()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalPoint$, a realization the distribution
    \end{description}

  \item $getReferenceBandwidth$
    \begin{description}
    \item[Usage:]  $getReferenceBandwidth()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalPoint$, the bandwidth used for $PDF$
    \end{description}

  \item $getSample$
    \begin{description}
    \item[Usage:]  $getSample()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $n$: a positive integer
    \end{description}
    \item[Value:] a $NumericalSample$, a sample of $n$ realizations of the distribution
    \end{description}

  \item $getStandardDeviation$
    \begin{description}
    \item[Usage:]  $getStandardDeviation()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $None$
    \end{description}
    \item[Value:] a $NumericalPoint$, the standard deviation of the distribution
    \end{description}

  \item $setAlpha$
    \begin{description}
    \item[Usage:]  $setAlpha(alpha)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $alpha$, a $NumericalScalar$
    \end{description}
    \item[Value:]  $None$, sets the $\alpha$ parameter used for the computation of the range (position indicator)
    \end{description}

  \item $setBeta$
    \begin{description}
    \item[Usage:] $setBeta(beta)$
    \item[Arguments:] \strut
    \begin{description}
      \item $beta$: a $NumericalScalar$
    \end{description}
    \item[Value:] $None$, sets the $\beta$ parameter used for the computation of the range (dispersion indicator)
    \end{description}

  \item $setBlockMax$
    \begin{description}
    \item[Usage:]  $setBlockMax(blocMax)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $blocMax$, an $UnsignedLong$
    \end{description}
    \item[Value:] $None$, sets the maximal parameter size (i.e $2^{blocMax}$) used for the computation of the characteristic function differences
    \end{description}

  \item $setBlockMin$
    \begin{description}
    \item[Usage:]  $setBlockMin(blocMin)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $blocMin$, an $UnsignedLong$
    \end{description}
    \item[Value:] $None$, sets the minimal parameter size (i.e $2^{blocMin}$) used for the computation of the characteristic function differences
    \end{description}

  \item $setMaxSize$
    \begin{description}
    \item[Usage:]  $setMaxSize(maxSize)$
    \item[Arguments:]  \strut
    \begin{description}
      \item $maxSize$, an $UnsignedLong$
    \end{description}
    \item[Value:] $None$, sets the maximum size of the cache for the CharacteristicFunction values
    \end{description}

  \item $setReferenceBandwidth$
    \begin{description}
    \item[Usage:]  $setReferenceBandwidth(bandwidth)$
    \item[Arguments:] \strut
    \begin{description}
      \item $bandwidth$, a $NumericalPoint$
    \end{description}
    \item[Value:] $None$, sets the bandwidth used for $PDF$
    \end{description}

  \item $setPDFPrecision$
    \begin{description}
    \item[Usage:]  $setPDFPrecision()$
    \item[Arguments:]  \strut
    \begin{description}
      \item $pdfPrecision$, a $NumericalScalar$
    \end{description}
    \item[Value:] $None$, sets the precision used for the computation of the probability density function
    \end{description}

  \end{description}

\end{description}

\subsection{MultivariateRandomMixture}
This class inherits from the $Distribution$ class.

\begin{description}
\item[Usage:]  \strut
  \begin{description}
  \item $MultivariateRandomMixture(collection, matrix)$
  \item $MultivariateRandomMixture(collection, matrix, vector)$
  \end{description}
\item[Arguments:] \strut
  \begin{description}
  \item $collection$: a DistributionCollection, a collection of OpenTURNS distributions.
  \item $matrix$: a Matrix, the matrix of the affine transformation.
  \item $vector$: a NumericalPoint, the constant term of the affine transformation.
  \end{description}
\end{description}
